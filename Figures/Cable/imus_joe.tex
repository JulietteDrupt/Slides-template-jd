    \def\unitl{0.8}
    \def\angleAB{-30}
    \def\angleBR{45}

    \coordinate (A) at ($(O) + (0,-1.5)$);
    \coordinate (B) at ($(A) + (\angleAB:3)$);
    \coordinate (R) at ($(B) + (\angleBR:2.5)$);

    % Cable + ROV + ballast
    \draw[very thick, orange] (O) -- (A) -- (B) -- (R) -- (B);
    \draw[thick, black] (R) rectangle ($(R) + (0.5,0.3)$);
    \draw[thick, black] ($(B) - (0.1,0)$) rectangle ($(B) + (0.1,-0.2)$);

    % Points
    \filldraw (O) circle (1pt) node[above=3pt] {$\mathbf{O}$};
    \filldraw (A) circle (1pt) node[left] {$\mathbf{A}$};
    \node[below=0.2cm] at (B) {$\mathbf{B}$};
    \filldraw (R) circle (1pt) node[left, yshift=0.2cm] {$\mathbf{R}$};

    % IMU 1
    \coordinate (I1) at ($ (A)!.4!(B) $);
    \draw[thick, blue!50, fill=blue!5, rotate around = {\angleAB:(I1)}] (I1) rectangle ($(I1) + (0.5,-0.5)$);

    \filldraw (I1) circle (1pt) node[above] {$\mathbf{I}_1$};

    \draw[-latex, thick, green, rotate around = {\angleAB:(I1)}] ($(I1) + (0.25,-0.25)$) -- ($(I1) + (0.25,-0.25) + (0, -\unitl)$) node[right, black] {$\mathcal{F}_{\mathbf{I}_1}$};
    \draw[blue, thick, rotate around = {\angleAB:(I1)}] ($(I1) + (0.25,-0.25)$) circle (5pt);
    \draw ($(I1) + (0.25,-0.25)$) node[cross=4pt, blue, thick, rotate around = {\angleAB:(I1)}] {};

    % IMU 2
    \coordinate (I2) at ($ (R)!.4!(B) $);
    \draw[thick, blue!50, fill=blue!5, rotate around = {\angleBR:(I2)}] (I2) rectangle ($(I2) + (-0.5,-0.5)$);
    \draw[-latex, thick, purple, rotate around = {\angleBR:(I2)}] (I2) -- ($(I2) + (-\unitl, 0)$) node[above, yshift=0.15cm] {$\mathbf{v}_{i_2}$};
    \filldraw (I2) circle (1pt) node[above, xshift=-0.1cm] {$\mathbf{I}_2$};
    \node[below, xshift=0.1cm] at ($ (R)!.3!(I2) $) {$l_{i_2}$};

    \draw[-latex, thick, red, rotate around = {\angleBR:(I2)}] ($(I2) + (-0.25,-0.25)$) -- ($(I2) + (-0.25,-0.25) + (-\unitl, 0)$);
    \draw[-latex, thick, green, rotate around = {\angleBR:(I2)}] ($(I2) + (-0.25,-0.25)$) -- ($(I2) + (-0.25,-0.25) + (0, -\unitl)$) node[left, black] {$\mathcal{F}_{\mathbf{I}_2}$};
    \draw[blue, thick, rotate around = {\angleBR:(I2)}] ($(I2) + (-0.25,-0.25)$) circle (5pt);
    \draw[blue, thick, fill=blue, rotate around = {\angleBR:(I2)}] ($(I2) + (-0.25,-0.25)$) circle (1.5pt);

    \draw[-latex, thick, red] (O) -- ($(O) + (\unitl,0)$) node[above, black] {$\mathcal{F}_{\mathbf{O}}$};

    \draw[green, thick] (O) circle (5pt);
    \draw[green, thick, fill=green] (O) circle (1.5pt);

    \draw[-latex, thick, blue] (Pw) -- ($(Pw) + (0,\unitl)$);
